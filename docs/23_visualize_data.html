Site location

Code 
loc_plots <- read.csv("data/cache/data_stocks.csv") |>
  select(site, plot, lat, lon) |>
  drop_na()
leaflet() |>
  addProviderTiles(providers$Esri.WorldImagery) |>
  addMarkers(
    lng = loc_plots$lon, lat = loc_plots$lat,
    label = paste(loc_plots$site, loc_plots$plot)
  )
