Site location

Code 
loc_plots <- read.csv("data/plot_0.2.2.csv") |> 
  select(site, plot, lat, lon) |> 
  drop_na()
leaflet() |>
  addProviderTiles(providers$Esri.WorldImagery) |>
  addMarkers(lng=loc_plots$lon, lat=loc_plots$lat, 
             label = paste(loc_plots$site, loc_plots$plot))
